using System;
using System.Linq;

public class Program
{
	public static void Main()
	{
		bool result = true;
		int n = 9;
		int[] numbers = {2, 5, 6, 8};
		while (numbers.Max() > n)
		{
			break;
		}

		{
			if (numbers.Max() < n)
				result = false;
		}

		{
			Console.WriteLine(result);
		}
	}
}
